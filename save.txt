
Fazes com um `GtkWindow` modal sem decoração + `GtkBox` vertical.

### C

```c
static void
close_dialog(GtkButton *button, gpointer user_data)
{
  GtkWindow *dialog = GTK_WINDOW(user_data);
  gtk_window_close(dialog);
}

void
show_add_device_dialog(ui_t *ui)
{
  GtkWidget *dialog = gtk_window_new();

  gtk_window_set_modal(GTK_WINDOW(dialog), TRUE);
  gtk_window_set_transient_for(GTK_WINDOW(dialog), GTK_WINDOW(ui->window));
  gtk_window_set_default_size(GTK_WINDOW(dialog), 520, 420);
  gtk_window_set_decorated(GTK_WINDOW(dialog), FALSE);

  gtk_widget_add_css_class(dialog, "modal-window");

  GtkWidget *root = gtk_box_new(GTK_ORIENTATION_VERTICAL, 0);
  gtk_widget_add_css_class(root, "modal-card");
  gtk_window_set_child(GTK_WINDOW(dialog), root);

  GtkWidget *header = gtk_box_new(GTK_ORIENTATION_HORIZONTAL, 0);
  gtk_widget_add_css_class(header, "modal-header");

  GtkWidget *title = gtk_label_new("Add Network Device");
  gtk_widget_add_css_class(title, "modal-title");
  gtk_widget_set_hexpand(title, TRUE);
  gtk_label_set_xalign(GTK_LABEL(title), 0.0);

  GtkWidget *close_button = gtk_button_new_with_label("×");
  gtk_widget_add_css_class(close_button, "modal-close-button");

  g_signal_connect(close_button, "clicked", G_CALLBACK(close_dialog), dialog);

  gtk_box_append(GTK_BOX(header), title);
  gtk_box_append(GTK_BOX(header), close_button);

  GtkWidget *form = gtk_box_new(GTK_ORIENTATION_VERTICAL, 12);
  gtk_widget_add_css_class(form, "modal-content");

  gtk_box_append(GTK_BOX(form), gtk_entry_new());
  gtk_box_append(GTK_BOX(form), gtk_entry_new());
  gtk_box_append(GTK_BOX(form), gtk_entry_new());

  GtkWidget *actions = gtk_box_new(GTK_ORIENTATION_HORIZONTAL, 8);
  gtk_widget_add_css_class(actions, "modal-actions");
  gtk_widget_set_halign(actions, GTK_ALIGN_END);

  GtkWidget *cancel = gtk_button_new_with_label("Cancel");
  GtkWidget *add = gtk_button_new_with_label("Add");

  gtk_widget_add_css_class(cancel, "modal-cancel-button");
  gtk_widget_add_css_class(add, "modal-add-button");

  g_signal_connect(cancel, "clicked", G_CALLBACK(close_dialog), dialog);

  gtk_box_append(GTK_BOX(actions), cancel);
  gtk_box_append(GTK_BOX(actions), add);

  gtk_box_append(GTK_BOX(root), header);
  gtk_box_append(GTK_BOX(root), form);
  gtk_box_append(GTK_BOX(root), actions);

  gtk_window_present(GTK_WINDOW(dialog));
}
```
